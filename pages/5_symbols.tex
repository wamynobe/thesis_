\section*{Symbols and Quantities}
\noindent
\begin{tabular}{>{\raggedright\arraybackslash}p{0.24\textwidth} r c >{\raggedright\arraybackslash}p{0.50\textwidth}}
  $a_t$        & $[\cdot]$      & -- & attention weight assigned to frame $t$ \\
  $F_0$        & $[\text{Hz}]$  & -- & fundamental frequency \\
  $h_{l,t}$    & $[\cdot]$      & -- & hidden state from SSL layer $l$ at frame $t$ \\
  $h_t$        & $[\cdot]$      & -- & weighted frame representation at frame $t$ \\

  $T1$--$T4$, $T_{\mathrm{NS}}$ & $[\cdot]$  & -- & four lexical Mandarin tones and the non-standard learner-production category \\
  $z$          & $[\cdot]$      & -- & utterance-level vector after attention pooling \\
  $z_{t,k}$    & $[\cdot]$      & -- & frame-level logit for frame $t$ and class $k$ in the AISHELL-1 objective \\
  $\alpha_l$   & $[\cdot]$      & -- & softmax-normalised weight assigned to SSL layer $l$ \\
  $\gamma$     & $[\cdot]$      & -- & focal-loss focusing parameter \\
  $\theta$     & $[\cdot]$      & -- & trainable parameters of the model $f_{\theta}$ \\
\end{tabular}
